        async function loadData() {

            // On charge le fichier data.json qui sera généré par notre script Python

            const response = await fetch('data.json');

            const data = await response.json();

            

            // Mise à jour des KPIs

            document.getElementById('avg').innerText = data.stats.avg.toFixed(1);

            document.getElementById('max').innerText = data.stats.max;

            document.getElementById('tir').innerText = data.stats.tir.toFixed(1);


            // Graphique

            const ctx = document.getElementById('cgmChart').getContext('2d');

            new Chart(ctx, {

                type: 'line',

                data: {

                    datasets: [{

                        label: 'Glycémie (mg/dL)',

                        data: data.points.map(p => ({ x: p.t, y: p.v })),

                        borderColor: '#58a6ff',

                        backgroundColor: 'rgba(88, 166, 255, 0.1)',

                        fill: true,

                        tension: 0.3,

                        pointRadius: 0

                    }]

                },

                options: {

                    responsive: true,

                    maintainAspectRatio: false,

                    scales: {

                        x: { type: 'time', time: { unit: 'day' }, grid: { color: '#30363d' } },

                        y: { min: 40, max: 450, grid: { color: '#30363d' } }

                    },

                    plugins: {

                        legend: { display: false }

                    }

                }

            });

        }

        loadData();

    </script>
